CREATE DATABASE colcheia_pontuada;
USE colcheia_pontuada;

CREATE TABLE formacao (
    id_formacao INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(50) NOT NULL
);

CREATE TABLE instrumento (
    id_instrumento INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(50),
    Familia SET('corda_friccionada','corda_pincada','corda_batida','sopro_madeira','sopro_metal','percussao_definida','percussao_indefinida','teclado','eletrônico')
);

CREATE TABLE maestro (
    id_maestro INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(50) NOT NULL,
    Espec SET('regente_orquestra','regente_coro','regente_opera','regente_banda','regente_camara','regente_juvenil','regente_militar','regente_contemporaneo','regente_barroco','regente_romantico')
);

CREATE TABLE musico (
    id_musico INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(50) NOT NULL,
    Nivel SET('Iniciante','Amador','Profissional')
);

CREATE TABLE grupo (
    id_grupo INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(50) NOT NULL,
    Nivel SET('Iniciante','Amador','Profissional'),
    id_formacao INT NOT NULL,
    id_maestro INT NOT NULL,
    CONSTRAINT fk_grupo_formacao FOREIGN KEY (id_formacao) REFERENCES formacao(id_formacao),
    CONSTRAINT fk_grupo_maestro FOREIGN KEY (id_maestro) REFERENCES maestro(id_maestro)
);

CREATE TABLE repertorio (
    id_repertorio INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(50) NOT NULL,
    id_grupo INT NOT NULL,
    CONSTRAINT fk_repertorio_grupo FOREIGN KEY (id_grupo) REFERENCES grupo(id_grupo)
);

CREATE TABLE peca (
    id_peca INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(50),
    id_formacao INT NOT NULL,
    bpm INT NOT NULL,
    tonalidade SMALLINT NOT NULL,
    CONSTRAINT fk_peca_formacao FOREIGN KEY (id_formacao) REFERENCES formacao(id_formacao)
);

CREATE TABLE grupo_musico (
    id_grupo_musico INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    id_grupo INT NOT NULL,
    id_musico INT NOT NULL,
    CONSTRAINT fk_grupo_musico_grupo FOREIGN KEY (id_grupo) REFERENCES grupo(id_grupo),
    CONSTRAINT fk_grupo_musico_musico FOREIGN KEY (id_musico) REFERENCES musico(id_musico),
    UNIQUE KEY uk_grupo_musico (id_grupo, id_musico)
);

CREATE TABLE repertorio_peca (
    id_repertorio_peca INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    id_repertorio INT NOT NULL,
    id_peca INT NOT NULL,
    CONSTRAINT fk_repertorio_peca_repertorio FOREIGN KEY (id_repertorio) REFERENCES repertorio(id_repertorio),
    CONSTRAINT fk_repertorio_peca_peca FOREIGN KEY (id_peca) REFERENCES peca(id_peca),
    UNIQUE KEY uk_repertorio_peca (id_repertorio, id_peca)
);

CREATE TABLE partitura (
    id_partitura INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    id_peca INT NOT NULL,
    id_instrumento INT NOT NULL,
    pdf LONGBLOB,
    CONSTRAINT fk_partitura_peca FOREIGN KEY (id_peca) REFERENCES peca(id_peca),
    CONSTRAINT fk_partitura_instrumento FOREIGN KEY (id_instrumento) REFERENCES instrumento(id_instrumento),
    UNIQUE KEY uk_partitura (id_peca, id_instrumento)
);

CREATE TABLE musico_partitura (
    id_musico_partitura INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    id_musico INT NOT NULL,
    id_partitura INT NOT NULL,
    CONSTRAINT fk_musico_partitura_musico FOREIGN KEY (id_musico) REFERENCES musico(id_musico),
    CONSTRAINT fk_musico_partitura_partitura FOREIGN KEY (id_partitura) REFERENCES partitura(id_partitura),
    UNIQUE KEY uk_musico_partitura (id_musico, id_partitura)
);

CREATE TABLE instrumento_formacao (
    id_instrumento_formacao INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    id_instrumento INT NOT NULL,
    id_formacao INT NOT NULL,
    CONSTRAINT fk_inst_form_instrumento FOREIGN KEY (id_instrumento) REFERENCES instrumento(id_instrumento),
    CONSTRAINT fk_inst_form_formacao FOREIGN KEY (id_formacao) REFERENCES formacao(id_formacao),
    UNIQUE KEY uk_instrumento_formacao (id_instrumento, id_formacao)
);

CREATE TABLE instrumento_musico (
    id_instrumento_musico INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    id_instrumento INT NOT NULL,
    id_musico INT NOT NULL,
    CONSTRAINT fk_inst_mus_instrumento FOREIGN KEY (id_instrumento) REFERENCES instrumento(id_instrumento),
    CONSTRAINT fk_inst_mus_musico FOREIGN KEY (id_musico) REFERENCES musico(id_musico),
    UNIQUE KEY uk_instrumento_musico (id_instrumento, id_musico)
);
